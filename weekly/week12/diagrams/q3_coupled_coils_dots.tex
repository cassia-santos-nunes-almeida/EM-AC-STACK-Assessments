% Q3: Coupled coils with dot convention — both i enter dotted terminals (aiding)
\documentclass[border=10pt]{standalone}
\usepackage[american voltages, american currents]{circuitikz}
\usepackage{amsmath}
\renewcommand{\familydefault}{\sfdefault}

\begin{document}
\begin{circuitikz}[
    line width=0.8pt,
    every node/.style={font=\sffamily}
]

% Coil 1 (left)
\draw (0,3) to[L, l=$L_1$] (0,0);
% Dot on coil 1 (top terminal = dotted)
\fill (0.25,2.8) circle (3pt);
\node[left] at (-0.3,2.8) {\textbullet};

% Coil 2 (right)
\draw (4,3) to[L, l=$L_2$] (4,0);
% Dot on coil 2 (top terminal = dotted)
\fill (3.75,2.8) circle (3pt);

% Mutual inductance coupling symbol (dashed arc)
\draw[<->, >=stealth, dashed, thick] (0.6,1.5) -- (3.4,1.5)
    node[midway, above] {$M$};

% Current i1 entering dotted terminal (top) of coil 1
\draw[->, >=stealth, thick, red!70!black] (-1.5,3.3) -- (-0.1,3.3)
    node[midway, above] {$i_1$};
\draw (-1.5,3.3) -- (-1.5,3) -- (0,3);

% Current i2 entering dotted terminal (top) of coil 2
\draw[->, >=stealth, thick, blue!70!black] (5.5,3.3) -- (4.1,3.3)
    node[midway, above] {$i_2$};
\draw (5.5,3.3) -- (5.5,3) -- (4,3);

% Voltage labels
\node[left] at (-0.3,1.5) {$v_1$};
\draw[->, >=stealth] (-0.5,2.5) -- (-0.5,0.5);

\node[right] at (4.3,1.5) {$v_2$};
\draw[->, >=stealth] (4.5,2.5) -- (4.5,0.5);

% Bottom terminals
\draw (-1.5,0) -- (0,0);
\draw (4,0) -- (5.5,0);

% Terminal dots
\fill (0,3) circle (2pt);
\fill (0,0) circle (2pt);
\fill (4,3) circle (2pt);
\fill (4,0) circle (2pt);

% Coupling coefficient label
\node[below] at (2,0) {Coupling coefficient $k$};

\end{circuitikz}
\end{document}
